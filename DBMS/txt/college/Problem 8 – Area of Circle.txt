-- Step 1: Create and select database
CREATE DATABASE IF NOT EXISTS circle_db;
USE circle_db;

-- Step 2: Create the table
DROP TABLE IF EXISTS areas;
CREATE TABLE areas (
    radius FLOAT,
    area   FLOAT
);

-- Step 3: Create a stored procedure to calculate and insert areas
DELIMITER $$

CREATE PROCEDURE calc_areas()
BEGIN
    DECLARE r FLOAT DEFAULT 5;
    DECLARE a FLOAT;
    DECLARE pi FLOAT DEFAULT 3.14159;

    WHILE r <= 9 DO
        SET a = pi * r * r;
        INSERT INTO areas VALUES (r, a);
        SET r = r + 1;
    END WHILE;
END $$

DELIMITER ;

-- Step 4: Execute the procedure
CALL calc_areas();

-- Step 5: Display the results
SELECT * FROM areas;
